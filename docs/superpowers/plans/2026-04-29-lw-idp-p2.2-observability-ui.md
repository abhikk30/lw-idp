# lw-idp P2.2 — observability in the IDP UI implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Spec:** `docs/superpowers/specs/2026-04-29-lw-idp-p2.2-observability-ui-design.md`

**Goal:** Make the service detail Overview tab the canonical pane for "what's happening with this service right now" — logs (Loki), traces (Tempo), metrics (Prometheus), pod status (Kubernetes API) — all linked by `trace_id` and with deep-links to Grafana.

**Architecture:** Gateway grows a thin observability proxy plugin (`/api/v1/observability/*` + `/api/v1/services/<id>/pods`). Each endpoint resolves the catalog row's `targetNamespace` server-side and issues a single read to Loki/Tempo/Prometheus/k8s API. The web Overview tab is rebuilt as a 4-panel client-side layout that consumes those endpoints with React Query polling. Cross-linking lives entirely in UI state.

**Tech stack:** TypeScript end-to-end. Backend: Fastify plugins, axios for upstream HTTP, vitest for tests. Frontend: Next.js client components, React Query for polling, recharts for sparklines, shadcn/ui drawer for the trace view.

**Branch:** `p2.2-observability-ui` (off master, already created at `642135b`). Single PR titled `p2.2: observability in the IDP UI` at end.

```ts
// packages/service-kit/src/observability/loki.ts
import type { Logger } from "pino";

export interface LokiLine {
  ts: string;
  raw: string;
  level: string | null;
  msg: string | null;
  trace_id: string | null;
  span_id: string | null;
  pod: string | null;
}

export interface LokiQueryRangeOpts {
  query: string;        // LogQL
  startNs: bigint;      // RFC3339 nanos
  endNs: bigint;
  limit: number;
  direction?: "forward" | "backward";  // default backward (newest-first)
}

export interface LokiClient {
  queryRange(opts: LokiQueryRangeOpts): Promise<{ lines: LokiLine[] }>;
}

export class LokiError extends Error {
  constructor(public readonly status: number, msg: string) { super(msg); }
}

export function createLokiClient(opts: { baseUrl: string; logger?: Logger }): LokiClient {
  return {
    async queryRange(q) {
      const url = new URL("/loki/api/v1/query_range", opts.baseUrl);
      url.searchParams.set("query", q.query);
      url.searchParams.set("start", q.startNs.toString());
      url.searchParams.set("end", q.endNs.toString());
      url.searchParams.set("limit", String(q.limit));
      url.searchParams.set("direction", q.direction ?? "backward");

      const res = await fetch(url, { method: "GET" });
      if (!res.ok) {
        const body = await res.text().catch(() => "");
        throw new LokiError(res.status, `loki query failed: ${res.status} ${body.slice(0, 200)}`);
      }
      const json = (await res.json()) as LokiQueryRangeResponse;
      return { lines: parseLokiResponse(json) };
    },
  };
}

interface LokiQueryRangeResponse {
  data: { result: Array<{ stream: Record<string, string>; values: Array<[string, string]> }> };
}

function parseLokiResponse(r: LokiQueryRangeResponse): LokiLine[] {
  const out: LokiLine[] = [];
  for (const stream of r.data.result) {
    for (const [tsNs, raw] of stream.values) {
      out.push(parseLine(tsNs, raw, stream.stream));
    }
  }
  return out;
}

function parseLine(tsNs: string, raw: string, streamLabels: Record<string, string>): LokiLine {
  const ts = new Date(Number(BigInt(tsNs) / 1_000_000n)).toISOString();
  let parsed: Record<string, unknown> | null = null;
  try { parsed = JSON.parse(raw) as Record<string, unknown>; } catch { /* not JSON */ }
  const str = (k: string) => (parsed && typeof parsed[k] === "string" ? (parsed[k] as string) : null);
  return {
    ts,
    raw,
    level: str("level"),
    msg: str("msg"),
    trace_id: str("trace_id") ?? str("traceId"),
    span_id: str("span_id") ?? str("spanId"),
    pod: streamLabels.pod ?? null,
  };
}
```

```ts
// packages/service-kit/src/observability/tempo.ts
export interface TraceSummary {
  trace_id: string;
  root_service: string;
  root_operation: string;
  started_at: string;
  duration_ms: number;
  span_count: number;
  status: "ok" | "error";
}

export interface SpanNode {
  span_id: string;
  parent_id: string | null;
  service: string;
  name: string;
  started_at: string;
  duration_ms: number;
  status: "ok" | "error";
  attributes: Record<string, string | number | boolean>;
}

export interface TempoClient {
  searchTraces(opts: { serviceName: string; sinceMs: number; limit: number }): Promise<TraceSummary[]>;
  getTrace(traceId: string): Promise<{ trace_id: string; spans: SpanNode[] } | null>;
}

export class TempoError extends Error {
  constructor(public readonly status: number, msg: string) { super(msg); }
}

export function createTempoClient(opts: { baseUrl: string }): TempoClient {
  return {
    async searchTraces({ serviceName, sinceMs, limit }) {
      const url = new URL("/api/search", opts.baseUrl);
      url.searchParams.set("q", `{ resource.service.name="${serviceName}" }`);
      url.searchParams.set("limit", String(limit));
      url.searchParams.set("start", String(Math.floor((Date.now() - sinceMs) / 1000)));
      url.searchParams.set("end", String(Math.floor(Date.now() / 1000)));
      const res = await fetch(url);
      if (!res.ok) throw new TempoError(res.status, `tempo search failed: ${res.status}`);
      const json = (await res.json()) as TempoSearchResponse;
      return (json.traces ?? []).map(parseSummary);
    },
    async getTrace(traceId) {
      const url = new URL(`/api/traces/${encodeURIComponent(traceId)}`, opts.baseUrl);
      const res = await fetch(url);
      if (res.status === 404) return null;
      if (!res.ok) throw new TempoError(res.status, `tempo get-trace failed: ${res.status}`);
      const json = (await res.json()) as TempoGetTraceResponse;
      return { trace_id: traceId, spans: parseSpans(json) };
    },
  };
}

interface TempoSearchResponse { traces?: Array<TempoTraceResult> }
interface TempoTraceResult { traceID: string; rootServiceName: string; rootTraceName: string; startTimeUnixNano: string; durationMs: number; spanSet?: { spans: Array<{ status?: string }> } }
interface TempoGetTraceResponse { batches: Array<{ resource: { attributes: Array<{ key: string; value: { stringValue?: string } }> }; scopeSpans: Array<{ spans: Array<TempoApiSpan> }> }> }
interface TempoApiSpan {
  spanId: string;
  parentSpanId?: string;
  name: string;
  startTimeUnixNano: string;
  endTimeUnixNano: string;
  status?: { code?: number };
  attributes: Array<{ key: string; value: { stringValue?: string; intValue?: string; boolValue?: boolean } }>;
}

function parseSummary(t: TempoTraceResult): TraceSummary {
  const startMs = Number(BigInt(t.startTimeUnixNano) / 1_000_000n);
  const errSpan = (t.spanSet?.spans ?? []).find((s) => s.status === "STATUS_CODE_ERROR");
  return {
    trace_id: t.traceID,
    root_service: t.rootServiceName,
    root_operation: t.rootTraceName,
    started_at: new Date(startMs).toISOString(),
    duration_ms: t.durationMs,
    span_count: (t.spanSet?.spans ?? []).length,
    status: errSpan ? "error" : "ok",
  };
}

function parseSpans(r: TempoGetTraceResponse): SpanNode[] {
  const spans: SpanNode[] = [];
  for (const batch of r.batches ?? []) {
    const svc = (batch.resource.attributes.find((a) => a.key === "service.name")?.value.stringValue) ?? "unknown";
    for (const ss of batch.scopeSpans ?? []) {
      for (const s of ss.spans ?? []) {
        const startMs = Number(BigInt(s.startTimeUnixNano) / 1_000_000n);
        const endMs = Number(BigInt(s.endTimeUnixNano) / 1_000_000n);
        const attrs: Record<string, string | number | boolean> = {};
        for (const a of s.attributes ?? []) {
          if (a.value.stringValue !== undefined) attrs[a.key] = a.value.stringValue;
          else if (a.value.intValue !== undefined) attrs[a.key] = Number(a.value.intValue);
          else if (a.value.boolValue !== undefined) attrs[a.key] = a.value.boolValue;
        }
        spans.push({
          span_id: s.spanId,
          parent_id: s.parentSpanId || null,
          service: svc,
          name: s.name,
          started_at: new Date(startMs).toISOString(),
          duration_ms: endMs - startMs,
          status: s.status?.code === 2 ? "error" : "ok",
          attributes: attrs,
        });
      }
    }
  }
  spans.sort((a, b) => a.started_at.localeCompare(b.started_at));
  return spans;
}
```

```ts
// packages/service-kit/src/observability/prometheus.ts
export interface PromPoint { ts: string; value: number }
export interface PromQueryRangeOpts { query: string; startMs: number; endMs: number; stepSec: number }
export interface PromClient { queryRange(o: PromQueryRangeOpts): Promise<{ points: PromPoint[] }> }
export class PromError extends Error {
  constructor(public readonly status: number, msg: string) { super(msg); }
}

export function createPromClient(opts: { baseUrl: string }): PromClient {
  return {
    async queryRange({ query, startMs, endMs, stepSec }) {
      const url = new URL("/api/v1/query_range", opts.baseUrl);
      url.searchParams.set("query", query);
      url.searchParams.set("start", String(startMs / 1000));
      url.searchParams.set("end", String(endMs / 1000));
      url.searchParams.set("step", `${stepSec}s`);
      const res = await fetch(url);
      if (!res.ok) throw new PromError(res.status, `prom query failed: ${res.status}`);
      const json = (await res.json()) as PromResponse;
      const series = json.data?.result?.[0]?.values ?? [];
      return {
        points: series.map(([secStr, valStr]) => ({
          ts: new Date(Number(secStr) * 1000).toISOString(),
          value: Number(valStr),
        })),
      };
    },
  };
}

interface PromResponse {
  status: string;
  data: { resultType: string; result: Array<{ metric: Record<string, string>; values: Array<[number, string]> }> };
}
```

```ts
// apps/gateway-svc/src/http/observability.ts (logs portion)
import type { FastifyPluginAsync } from "fastify";
import fp from "fastify-plugin";
import { createLokiClient, createTempoClient } from "@lw-idp/service-kit";
import type { CatalogClient } from "../clients/catalog.js";

export interface ObservabilityPluginOptions {
  catalogClient: CatalogClient;
  lokiUrl: string;
  tempoUrl: string;
  promUrl: string;
}

const obsPluginFn: FastifyPluginAsync<ObservabilityPluginOptions> = async (fastify, opts) => {
  const loki = createLokiClient({ baseUrl: opts.lokiUrl });
  const tempo = createTempoClient({ baseUrl: opts.tempoUrl });

  fastify.get("/api/v1/observability/logs", async (req, reply) => {
    if (!req.session) return reply.code(401).send({ code: "unauthorized", message: "auth required" });
    const q = req.query as Record<string, string | undefined>;
    const serviceId = q.service;
    if (!serviceId) return reply.code(400).send({ code: "bad_request", message: "service required" });

    const svc = await opts.catalogClient.getService({ id: serviceId });
    if (!svc.service) return reply.code(404).send({ code: "not_found", message: "service not found" });
    const ns = svc.service.targetNamespace ?? "";
    if (!ns) return reply.code(409).send({ code: "no_namespace", message: "service has no targetNamespace; deploys not configured" });

    const sinceMs = parseSinceMs(q.since ?? "1h");
    const limit = Math.min(Number(q.limit ?? 200) || 200, 1000);
    const traceFilter = q.trace_id ? ` | trace_id="${q.trace_id}"` : "";
    const query = `{namespace="${ns}"} | json${traceFilter}`;
    const now = Date.now();
    const startNs = BigInt(now - sinceMs) * 1_000_000n;
    const endNs = BigInt(now) * 1_000_000n;

    try {
      const res = await loki.queryRange({ query, startNs, endNs, limit, direction: "backward" });
      return reply.send({ lines: res.lines, truncated: res.lines.length >= limit });
    } catch (err) {
      fastify.log.error({ err }, "loki query failed");
      return reply.code(502).send({ code: "loki_unreachable", message: "logs backend unreachable" });
    }
  });

  // (traces routes inlined below — Step 4)
};

function parseSinceMs(since: string): number {
  const m = /^(\d+)(ms|s|m|h)$/.exec(since);
  if (!m) return 60 * 60 * 1000;
  const n = Number(m[1]);
  switch (m[2]) {
    case "ms": return n;
    case "s":  return n * 1000;
    case "m":  return n * 60 * 1000;
    case "h":  return Math.min(n * 60 * 60 * 1000, 24 * 60 * 60 * 1000);
  }
  return 60 * 60 * 1000;
}

export const observabilityPlugin = fp(obsPluginFn, { name: "lw-idp-observability" });
```

```ts
  const q = req.query as Record<string, string | undefined>;
  if (!q.service) return reply.code(400).send({ code: "bad_request", message: "service required" });
  const panel = q.panel ?? "";
  if (!(panel in PROMQL)) return reply.code(400).send({ code: "bad_request", message: "panel must be req_rate|error_rate|p95_latency" });

  const svc = await opts.catalogClient.getService({ id: q.service });

  const sinceMs = parseSinceMs(q.since ?? "1h");

  const now = Date.now();

  try {
    const result = await prom.queryRange({
      query: PROMQL[panel](ns),
      startMs: now - sinceMs,
      endMs: now,
      stepSec,
    });
    const unit = panel === "req_rate" ? "req/s" : panel === "error_rate" ? "ratio" : "ms";
```

```ts
import * as fs from "node:fs";
import * as https from "node:https";

export interface K8sPod {
  name: string;
  phase: string;
  ready: boolean;
  started_at: string | null;
  restart_count: number;
  node: string | null;
}

export interface K8sClient {
  listPods(namespace: string, labelSelector?: string): Promise<K8sPod[]>;
}

export function createK8sClient(opts?: { baseUrl?: string; bearerToken?: string; caPath?: string }): K8sClient {
  const baseUrl = opts?.baseUrl ?? `https://kubernetes.default.svc`;
  const token = opts?.bearerToken ?? readIfExists("/var/run/secrets/kubernetes.io/serviceaccount/token");
  const caPath = opts?.caPath ?? "/var/run/secrets/kubernetes.io/serviceaccount/ca.crt";
  const ca = readIfExists(caPath);
  const agent = ca ? new https.Agent({ ca }) : undefined;

  return {
    async listPods(namespace, labelSelector) {
      const url = new URL(`/api/v1/namespaces/${encodeURIComponent(namespace)}/pods`, baseUrl);
      if (labelSelector) url.searchParams.set("labelSelector", labelSelector);
      // @ts-expect-error -- node fetch accepts dispatcher in newer node
      const res = await fetch(url, {
        headers: token ? { authorization: `Bearer ${token}` } : {},
        ...(agent ? { dispatcher: agent } : {}),
      });
      if (!res.ok) throw new Error(`k8s list pods failed: ${res.status}`);
      const json = (await res.json()) as { items: Array<unknown> };
      return (json.items as Array<K8sPodApi>).map(toPod);
    },
  };
}

function readIfExists(p: string): string | undefined {
  try { return fs.readFileSync(p, "utf8"); } catch { return undefined; }
}

interface K8sPodApi {
  metadata: { name: string };
  spec: { nodeName?: string };
  status: { phase: string; startTime?: string; containerStatuses?: Array<{ ready: boolean; restartCount: number }> };
}

function toPod(p: K8sPodApi): K8sPod {
  const cs = p.status.containerStatuses ?? [];
  const ready = cs.length > 0 && cs.every((c) => c.ready);
  const restart_count = cs.reduce((acc, c) => acc + c.restartCount, 0);
  return {
    name: p.metadata.name,
    phase: p.status.phase,
    ready,
    started_at: p.status.startTime ?? null,
    restart_count,
    node: p.spec.nodeName ?? null,
  };
}
```

```tsx
            <div className="text-destructive text-sm">Trace unavailable</div>
          ) : (
            <ul className="space-y-1 font-mono text-xs">
              {(data?.spans ?? []).map((s) => (
                <li key={s.span_id} className="border-l-2 pl-2" style={{ marginLeft: depth(s, data?.spans ?? []) * 16 }}>
                  <span className={s.status === "error" ? "text-destructive" : ""}>{s.service}</span>
                  {" "}<span className="text-muted-foreground">{s.name}</span>{" "}
                  <span>{s.duration_ms}ms</span>
                </li>
              ))}
            </ul>
          )}
        </div>
      </SheetContent>
    </Sheet>
  );
}

function depth(s: { parent_id: string | null }, all: Array<{ span_id: string; parent_id: string | null }>): number {
  let d = 0;
  let cur = s.parent_id;
  while (cur) {
    d += 1;
    const p = all.find((x) => x.span_id === cur);
    if (!p) break;
    cur = p.parent_id;
  }
  return d;
}
```
