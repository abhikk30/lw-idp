# lw-idp P1.2 — Service Skeletons + Helm + Tilt + CI Build Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** `tilt up` produces six healthy app pods on the `lw-idp-dev` kind cluster (`web`, `gateway-svc`, `identity-svc`, `catalog-svc`, `cluster-svc`, `notification-svc`). Each exposes `/healthz`, `/readyz`, and `/metrics` and survives a rolling restart without 5xx. CI builds + signs + chart-lints + runs an e2e smoke that verifies all 6 pods come Ready.

**Architecture:** Five backend services share a new `@lw-idp/service-kit` package that wraps Fastify + pino + OTel + graceful shutdown into a single `startServer({ name, port })` call — each service's `src/index.ts` is ~5 lines. The `web` app is a Next.js 15 shell with one placeholder route and a Route Handler for `/api/healthz`. Helm uses a library chart (`charts/lw-idp-service`) with named templates; each service chart imports that library and supplies only its own `values.yaml`. An umbrella chart (`charts/lw-idp`) composes all six.

**Tech stack:** Fastify 5, `fastify-metrics` (Prometheus), pino 9, `@lw-idp/telemetry` (OTel from P1.1), Next.js 15 App Router, Helm library charts, Tilt 0.37, docker buildx (multi-arch arm64+amd64), ko-less Docker builds, GitHub Actions + `helm/kind-action`.

**Spec:** `docs/superpowers/specs/2026-04-20-lw-idp-foundation-design.md` §§ 6 + 9 + 11.
**Predecessor:** `docs/superpowers/plans/2026-04-20-lw-idp-p1.1-foundation-bootstrap.md` (merged, tag `p1.1-foundation`).

## Conventions

- Every new TypeScript package follows the same pattern as Plan 1.1: `extends: @lw-idp/config/tsconfig`, `rootDir: "."`, `outDir: "dist"`, `include: [src, test]`.
- Every service exposes **exactly three** operational HTTP endpoints: `GET /healthz` (liveness, always 200 if process alive), `GET /readyz` (readiness; 200 once NATS/DB probes pass, 503 otherwise), `GET /metrics` (Prometheus text exposition).
- Every service logs via `pino` with `{ service, env, pid, trace_id, span_id }` bound fields (from `@lw-idp/telemetry`).
- Every service respects `SIGTERM` with a 10s `preStop sleep 5` + graceful HTTP close.
- Every Helm chart sets `replicas: 2`, `topologySpreadConstraints`, `podAntiAffinity`, `PodDisruptionBudget minAvailable: 1`, readiness/liveness probes on `/readyz`/`/healthz`, `resources.requests/limits`.
- All commits use Conventional Commits (`feat(svc):`, `infra(chart):`, `test(e2e):`, etc.).

```ts
import { createLogger } from "@lw-idp/telemetry";
import Fastify, { type FastifyInstance } from "fastify";
import fastifyMetrics from "fastify-metrics";

export interface ReadyProbeResult {
  ok: boolean;
  name: string;
  reason?: string;
}

export type ReadyProbe = () => Promise<ReadyProbeResult>;

export interface BuildServerOptions {
  name: string;
  port: number;
  host?: string;
  version?: string;
  readyProbes?: ReadyProbe[];
}

export interface LwIdpServer {
  fastify: FastifyInstance;
  listen(): Promise<string>;
  close(): Promise<void>;
}

export async function buildServer(opts: BuildServerOptions): Promise<LwIdpServer> {
  const logger = createLogger({ service: opts.name });
  const fastify = Fastify({
    loggerInstance: logger,
    disableRequestLogging: false,
    trustProxy: true,
  });

  await fastify.register(fastifyMetrics, {
    endpoint: "/metrics",
    defaultMetrics: { enabled: true },
    routeMetrics: { enabled: true },
  });

  fastify.get("/healthz", async () => ({ status: "ok", service: opts.name }));

  fastify.get("/readyz", async (_req, reply) => {
    const probes = opts.readyProbes ?? [];
    const results = await Promise.all(
      probes.map(async (probe) => {
        try {
          return await probe();
        } catch (err) {
          return { ok: false, name: "unknown", reason: err instanceof Error ? err.message : String(err) };
        }
      }),
    );
    const notReady = results.filter((r) => !r.ok);
    if (notReady.length > 0) {
      return reply.code(503).send({ status: "not_ready", probes: results });
    }
    return { status: "ready", probes: results };
  });

  const host = opts.host ?? "0.0.0.0";

  return {
    fastify,
    listen: async () => fastify.listen({ port: opts.port, host }),
    close: async () => {
      await fastify.close();
    },
  };
}
```

- [ ] **Step 8: Create `packages/service-kit/src/shutdown.ts`**

```ts
import type { LwIdpServer } from "./server.js";

export function wireGracefulShutdown(server: LwIdpServer, timeoutMs = 10_000): void {
  const signals: NodeJS.Signals[] = ["SIGTERM", "SIGINT"];
  let shuttingDown = false;
  for (const sig of signals) {
    process.once(sig, async () => {
      if (shuttingDown) return;
      shuttingDown = true;
      server.fastify.log.info({ signal: sig }, "graceful shutdown");
      const timer = setTimeout(() => {
        server.fastify.log.error({ signal: sig, timeoutMs }, "shutdown timed out — force exit");
        process.exit(1);
      }, timeoutMs);
      timer.unref();
      try {
        await server.close();
        process.exit(0);
      } catch (err) {
        server.fastify.log.error({ err }, "error during close");
        process.exit(1);
      }
    });
  }
}
```

```dockerfile
COPY --from=builder --chown=app:app /repo/apps/${SERVICE}/package.json ./package.json

USER app
EXPOSE 4000
```

```ts
import { execa } from "execa";
import { afterAll, beforeAll, describe, expect, it } from "vitest";

// ... existing top-of-file imports kept ...

describe("lw-idp service healthz", () => {
  const services = [
    { name: "gateway-svc", expected: "gateway-svc", port: 14000 },
    { name: "identity-svc", expected: "identity-svc", port: 14001 },
    { name: "catalog-svc", expected: "catalog-svc", port: 14002 },
    { name: "cluster-svc", expected: "cluster-svc", port: 14003 },
    { name: "notification-svc", expected: "notification-svc", port: 14004 },
  ];

  interface PortForward { kill(): void }
  const pfs: PortForward[] = [];

  beforeAll(async () => {
    for (const { name, port } of services) {
      const child = execa("kubectl", ["-n", "lw-idp", "port-forward", `svc/${name}`, `${port}:80`]);
      pfs.push({ kill: () => { child.kill("SIGTERM"); } });
    }
    await new Promise((r) => setTimeout(r, 3_000));
  }, 30_000);

  afterAll(() => {
    for (const pf of pfs) pf.kill();
  });

  for (const { name, expected, port } of services) {
    it(`${name} /healthz responds 200 with correct service name`, async () => {
      const res = await fetch(`http://localhost:${port}/healthz`);
      expect(res.status).toBe(200);
      expect(await res.json()).toEqual({ status: "ok", service: expected });
    });
  }

  it("web /api/healthz responds 200", async () => {
    const child = execa("kubectl", ["-n", "lw-idp", "port-forward", "svc/web", "13001:80"]);
    try {
      await new Promise((r) => setTimeout(r, 2_000));
      const res = await fetch("http://localhost:13001/api/healthz");
      expect(res.status).toBe(200);
      expect(await res.json()).toEqual({ status: "ok", service: "web" });
    } finally {
      child.kill("SIGTERM");
    }
  });
});
```
